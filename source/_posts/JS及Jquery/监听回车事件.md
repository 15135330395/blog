## 页面上监听回车调用方法
 可用于 直接回车登陆 以及 直接回车查询
	document.onkeydown = function(){
		var event = window.event;
		if (event.keyCode == 13) {
			search();
		}
	};
	$(document).keydown(function(event){
		if(event.keyCode == 13){
			alert('你按下了Enter'); 
		}
	});
